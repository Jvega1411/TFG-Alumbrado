fb_Horario_2t
hora_actual:=  (int_to_dint(horplc)*3600)+(int_to_dint(minplc)*60);

horarios_inicio_1:= (int_to_dint(horarios.HORA_INICIO_1)*3600)+(INT_TO_DINT(horarios.MINUTO_INICIO_1)*60);
horarios_fin_1:= (int_to_dint(horarios.HORA_FIN_1)*3600)+(INT_TO_DINT(horarios.MINUTO_FIN_1)*60);

horarios_inicio_2:= (int_to_dint(horarios.HORA_INICIO_2)*3600)+(INT_TO_DINT(horarios.MINUTO_INICIO_2)*60);
horarios_fin_2:= (int_to_dint(horarios.HORA_FIN_2)*3600)+(INT_TO_DINT(horarios.MINUTO_FIN_2)*60);

horario_activo:= FALSE;

IF (horarios_inicio_1<>horarios_fin_1) THEN
	IF (horarios_fin_1>=horarios_inicio_1) THEN
		IF (hora_actual>=horarios_inicio_1) AND (hora_actual<horarios_fin_1) THEN
			horario_activo:= TRUE;
		END_IF;
	ELSE
		IF (hora_actual>=horarios_inicio_1) OR (hora_actual<horarios_fin_1) THEN
			horario_activo:= TRUE;
		END_IF;
	END_IF;
END_IF;
IF (horarios_inicio_2<>horarios_fin_2) THEN
	IF (horarios_fin_2>=horarios_inicio_2) THEN
		IF (hora_actual>=horarios_inicio_2) AND (hora_actual<horarios_fin_2) THEN
			horario_activo:= TRUE;
		END_IF;
	ELSE
		IF (hora_actual>=horarios_inicio_2) OR (hora_actual<horarios_fin_2) THEN
			horario_activo:= TRUE;
		END_IF;
	END_IF;
END_IF;
